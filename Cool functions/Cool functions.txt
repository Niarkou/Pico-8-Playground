-- Cool Outlined print

function coprint(text, x, y, color)
    for i = -1,1 do
        for j = -1,1 do
            print(text, x+i, y+j, color or 7)
end

-- Cool print (centered)

function cprint(text, y, color)
    local x = 64 - 2 * #text
    coprint(text, x, y, color or 7)
end

-- Cool rnd (random min and max)

function crnd(min, max)
    return min + rnd(max-min)
end
